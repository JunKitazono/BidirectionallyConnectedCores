1function x = baseMto10(sigma, M, N_st)
 2
 3%-------------------------------------------------------------------------------------------------
 4% PURPOSE: convert base M to base 10
 5%
 6% INPUTS:
 7%   sigma: base M value
 8%   M: base
 9%
10% OUTPUTS:
11%  x: decimal value
12%-------------------------------------------------------------------------------------------------
13%
14% Masafumi Oizumi, 2018
15
16assert( isa( sigma, 'double' ) );
17assert( isa( M, 'double' ) );
18assert( isa( N_st, 'double' ) );
19
20assert( all( size(sigma) == [N_st 1] ) );
21assert( all( size(N_st) == [1 1] ) );
22assert( all( size(M) == [N_st 1]) ); 
23
24N = length(sigma);
25
26x = 0;
27for i=1: N
28    x = x + M^(i-1)*sigma(i);
29end
