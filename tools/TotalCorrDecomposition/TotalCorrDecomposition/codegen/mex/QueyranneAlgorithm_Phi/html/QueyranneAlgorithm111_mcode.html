1function [ IndexOutput ] = QueyranneAlgorithm( F, index )
 2% [ IndexOutput ] = QueyranneAlgorithm( F, Index ) 
 3% 
 4% This function is an implementation of the Queyranne's algorithm for a
 5% symmetric submodular function F of the set Index.
 6%
 7% Input ---
 8%       F: A function hundle, F( X ) is supposed a symmetric submodular function of X.
 9%       Index: A set of indices implying the domain of the F.
10%
11% Output ---
12%       IndexOutput: the index set minimizing the function F.
13% 
14% First written by Shohei Hidaka   Oct 22nd, 2015.
15% Revised (added its description in comments) by Shohei Hidaka Jan 13th, 2016.
16% Revised (rename PendentPair and add one line for the preprocess of the index ) by Shohei Hidaka March 18th, 2016.
17% Revised (additional minor function to sort the output argument) by Shohei Hidaka on April 5th, 2016
18% Revised (additional minor part for input and output argument) by Shohei Hidaka on April 26th, 2016
19%
20% See also: MaxTotalCorr, PendentPair
21%
22% References: Queyranne, M. (1998). Minimizing symmetric submodular functions. Mathematical Programming, 82(1-2), 3-12.
23% http://link.springer.com/article/10.1007%2FBF01585863
24
25%% 
26assert( isa( F, 'function_handle' ) )
27assert( isa( index, 'cell' )||isa( index, 'double' ) )
28
29%% Added March 18th, 2016./ further revised on April 26th.
30if ~iscell( index )
31    index = arrayfun( @(x) x, 1:length( index ), 'UniformOutput', false );
32end
33%% Added March 18th, 2016.
34
35N = length( index );
36M = sum( cellfun( @numel, index ) ); % further revised on April 26th.
37
38indexrec = cell( 1, N - 1 );
39f = zeros( 1, N - 1 );
40for i = 1:( N - 1 )
41    %fprintf( 'i = %d: ', i );
42    [ pp ] = PendentPair( F, index, 1 );
43    indexrec( i ) = index( pp( end ) );
44    f( i ) = F( index( pp( end ) ) ); %TotalCorr( index( pp( end ) ), CV );
45    index = index( pp );
46    index = [ index( 1:end-2 ), { cell2mat( index( end-1:end ) ) } ];
47    %fprintf( 'F = %.3f\n', f( i ) );
48end
49[ tmp, minind ] = min( f );
50%% Revised on April 5th, 2016
51%IndexOutput = indexrec{ minind };
52% if numel( indexrec{ minind } ) > N/2
53%     IndexOutput = setdiff( 1:N, indexrec{ minind } );
54% else
55%     IndexOutput = sort( indexrec{ minind } );
56% end
57% further revised on April 26th.
58if ~isempty( f ) & numel( indexrec{ minind } ) > M/2
59    IndexOutput = setdiff( cell2mat( cellfun( @(x) x(:)', index, 'UniformOutput', false ) ), indexrec{ minind } );
60elseif ~isempty( f )
61    IndexOutput = sort( indexrec{ minind } );
62end
