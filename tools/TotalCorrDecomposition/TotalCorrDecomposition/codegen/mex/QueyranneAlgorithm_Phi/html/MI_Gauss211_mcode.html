1function MI = MI_Gauss(Cov_X,Cov_XY,Cov_Y,Z)
 2
 3%-----------------------------------------------------------------------
 4% FUNCTION: MI_Gauss.m
 5% PURPOSE:  calculate Multi (Mutual) Information given covariance of data
 6%
 7% (Ay, 2001; Ay, 2015, Entropy; Barrett & Seth, 2011, PLoS Comp Biol)  
 8%
 9% INPUTS:   
10%           Cov_X: covariance of data X (past, t-tau)
11%           Cov_XY: cross-covariance of X (past, t-tau) and Y (present, t)
12%           Cov_Y: covariance of data Y (present, t)
13%           Z: partition of each channel (default: atomic partition)
14%
15% OUTPUT:
16%           MI: multi interaction SI(Y|X) among subsystems
17%
18%  Jun Kitazono, 2017
19%-----------------------------------------------------------------------
20
21N = size(Cov_X,1); % number of channels
22if nargin < 3
23    Cov_Y = Cov_X;
24end
25if nargin < 4
26    Z = 1: 1: N;
27end
28
29H = H_gauss([Cov_X, Cov_XY; Cov_XY', Cov_Y]);
30
31%%
32N_c = max(Z); % number of clusters
33H_p = zeros(N_c,1);
34
35for i=1: N_c
36    M = find(Z==i);
37    Cov_X_p = Cov_X(M,M);
38    Cov_Y_p = Cov_Y(M,M);
39    Cov_XY_p = Cov_XY(M,M);
40    
41    Cov_p = [Cov_X_p, Cov_XY_p; Cov_XY_p', Cov_Y_p];
42    H_p(i) = H_gauss(Cov_p);
43end
44
45%% stochastic interaction
46MI = sum(H_p) - H;
47
48
49end
