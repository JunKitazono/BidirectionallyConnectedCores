1function [ pendantPair ] = PendentPair( F, index, ind )
 2%% [ pendantPair ] = PendentPair( F, index, ind ) 
 3%% 
 4%% This function is the core sub-routine of the QueyranneAlgorithm.m
 5%% 
 6%% First written by Shohei Hidaka   Oct 22nd, 2015.
 7%% Revised (added its description in comments) by Shohei Hidaka Jan 13th, 2016.
 8%%
 9%% See also: QueyranneAlgorithm
10%%
11%% References: Queyranne, M. (1998). Minimizing symmetric submodular functions. Mathematical Programming, 82(1-2), 3-12.
12%% http://link.springer.com/article/10.1007%2FBF01585863
13if nargin < 2
14    ind = 1;
15end
16
17for i = 1:( length( index ) - 1 )
18%     disp(ind)
19    indc = setdiff( 1:length( index ), ind );
20    candidates = index( indc );
21    
22    TCtemp = zeros( 1, length( candidates ) );
23    for j = 1:length( candidates )
24        TCtemp( j ) = F( [ index( ind ), candidates( j ) ] ) - F( [ candidates( j ) ] );
25    end
26    [ TCgreedy, minind ] = min( TCtemp );
27    ind = [ ind, indc( minind ) ];
28end
29pendantPair = ind;
