1function phi = phi_for_Queyranne( indices, Cov_X, Cov_XY, Cov_Y, type_of_phi )
 2%-----------------------------------------------------------------------
 3% FUNCTION: SI_Gauss.m
 4% PURPOSE:  calculate stochastic interaction given covariance of data
 5% (Ay, 2001; Ay, 2015, Entropy; Barrett & Seth, 2011, PLoS Comp Biol)
 6% Here we only consider bi-partitions and represent them by specifying
 7% indices of one of two subsystems. 
 8%
 9% INPUTS:   
10%           indices: indices of a subsystem
11%           Cov_X: covariance of data X (past, t-tau)
12%           Cov_XY: cross-covariance of X (past, t-tau) and Y (present, t)
13%           Cov_Y: covariance of data Y (present, t)
14%           type_of_phi:
15%              'SI': phi_H, stochastic interaction
16%              'Geo': phi_G, information geometry version
17%              'star': phi_star, based on mismatched decoding
18%              'MI': Multi (Mutual) information, I(X_1, Y_1; X_2, Y_2)
19%              'MI1': Multi (Mutual) information. I(X_1; X_2). (IIT1.0)
20%              
21%
22% OUTPUT:
23%           phi: amount of integrated information 
24%-----------------------------------------------------------------------
25
26N = size(Cov_X, 1);
27
28if iscell(indices)
29    ind = cell2mat(indices);
30else
31    ind = indices;
32end
33
34Z = ones(1, N);
35Z(ind) = 2;
36
37switch type_of_phi
38    case 'SI'
39        phi = SI_Gauss(Cov_X, Cov_XY, Cov_Y, Z);
40    case 'Geo'
41        phi = phi_G_Gauss(Cov_X, Cov_XY, Cov_Y, Z);
42    case 'star'
43        phi = phi_star_Gauss(Cov_X, Cov_XY, Cov_Y, Z);
44    case 'MI'
45        phi = MI_Gauss(Cov_X, Cov_XY, Cov_Y, Z);
46    case 'MI1'
47        phi = MI1_Gauss(Cov_X, Z);
48end
49
50end
