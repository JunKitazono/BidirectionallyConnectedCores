1function [ IndexOutput ] = QueyranneAlgorithm( F, index_input )
 2% [ IndexOutput ] = QueyranneAlgorithm( F, Index ) 
 3% 
 4% This function is an implementation of the Queyranne's algorithm for a
 5% symmetric submodular function F of the set Index.
 6%
 7% Input ---
 8%       F: A function hundle, F( X ) is supposed a symmetric submodular function of X.
 9%       Index: A set of indices implying the domain of the F.
10%
11% Output ---
12%       IndexOutput: the index set minimizing the function F.
13% 
14% First written by Shohei Hidaka   Oct 22nd, 2015.
15% Revised (added its description in comments) by Shohei Hidaka Jan 13th, 2016.
16% Revised (rename PendentPair and add one line for the preprocess of the index ) by Shohei Hidaka March 18th, 2016.
17% Revised (additional minor function to sort the output argument) by Shohei Hidaka on April 5th, 2016
18% Revised (additional minor part for input and output argument) by Shohei Hidaka on April 26th, 2016
19%
20% See also: MaxTotalCorr, PendentPair
21%
22% References: Queyranne, M. (1998). Minimizing symmetric submodular functions. Mathematical Programming, 82(1-2), 3-12.
23% http://link.springer.com/article/10.1007%2FBF01585863
24
25%% Added March 18th, 2016./ further revised on April 26th.
26if iscell( index_input )
27    index = index_input
28else
29%     index = arrayfun( @(x) x, 1:length( index ), 'UniformOutput', false );
30    index = cell(1,length(index_input));
31    for i = 1:length(index)
32        index{i} = i;
33    end
34end
35
36
37%% Added March 18th, 2016.
38
39N = length( index );
40M = sum( cellfun( @numel, index ) ); % further revised on April 26th.
41
42indexrec = cell( 1, N - 1 );
43f = zeros( 1, N - 1 );
44for i = 1:( N - 1 );
45    %fprintf( 'i = %d: ', i );
46    [ pp ] = PendentPair( F, index, 1 );
47    indexrec( i ) = index( pp( end ) );
48    f( i ) = F( index( pp( end ) ) ); %TotalCorr( index( pp( end ) ), CV );
49    index = index( pp );
50    index = [ index( 1:end-2 ), { cell2mat( index( end-1:end ) ) } ];
51    %fprintf( 'F = %.3f\n', f( i ) );
52end
53[ tmp, minind ] = min( f );
54%% Revised on April 5th, 2016
55%IndexOutput = indexrec{ minind };
56% if numel( indexrec{ minind } ) > N/2
57%     IndexOutput = setdiff( 1:N, indexrec{ minind } );
58% else
59%     IndexOutput = sort( indexrec{ minind } );
60% end
61% further revised on April 26th.
62if ~isempty( f ) & numel( indexrec{ minind } ) > M/2
63    IndexOutput = setdiff( cell2mat( cellfun( @(x) x(:)', index, 'UniformOutput', false ) ), indexrec{ minind } );
64elseif ~isempty( f )
65    IndexOutput = sort( indexrec{ minind } );
66end
